actor SuperShotgunGiver : ClassWeaponGiver Replaces SuperShotgun
{	
	states
	{
		Spawn:
			SGN2 A 0
			SGN2 A -1 A_CheckFlag("DROPPED", "Dropped")
			loop
		Pickup:
			TNT1 A 0 A_JumpIf(ACS_ExecuteWithResult(GIVEWEAPON, 1, 16, 0) == TRUE, "Stay")
			goto Remove
		Dropped:
			TNT1 A 0 A_SpawnItemEx("SuperShotgunGiver_Dropped", 0, 0, 0, velx, vely, velz, angle, SXF_ABSOLUTEVELOCITY | SXF_ABSOLUTEANGLE, 0)
			stop
	}
}

actor SuperShotgunGiver_Dropped : SuperShotgunGiver
{
	States
	{
		Spawn:
			SGN2 A 63000
			goto FadeAway
		Pickup:
			TNT1 A 0 A_JumpIf(ACS_ExecuteWithResult(GIVEWEAPON, 1, 8, 1) == TRUE, "Stay")
			goto Remove
	}
}


actor SM_SuperShotgun : Weapon
{
	Tag "Super Shotgun"
	Decal BulletChip
	obituary "%k turned %o's torso into a pile of red gruel."
	radius 20
	height 16
	weapon.selectionorder 400
	weapon.kickback 100
	Weapon.AmmoType "SpreadAmmo"
	weapon.ammouse 2
	weapon.ammogive 0
	states
	{
		Ready:
			SHT2 A 1 A_WeaponReady
			goto Ready
		Deselect:
			SHT2 A 1 A_Lower
			goto Deselect
		Select:
			SHT2 A 1 A_Raise
			goto Select
		Fire:
			SHT2 A 0 radius_quake(4, 3, 0, 1, 0)
			SHT2 A 0 A_PlayWeaponSound("weapons/sshotf")
			SHT2 A 0 A_FireBullets (10, 6, 20, 5, "SSGPuff")
			SHT2 A 1 A_GunFlash
			SHT2 B 0 A_SetPitch(pitch-4)
			SHT2 BCDEFGHIJK 1 A_SetPitch(pitch+0.4)
			SHT2 A 1 A_CheckReload
			SH2R AB 2
			SH2R C 0 A_PlaySoundEx("weapons/sshoto", "auto", 0)
			SH2R CDEF 2
			//SH2R GG 0 A_FireCustomMissile("SSGCasing", Random(-14,8),0,2,0)
			SH2R GHI 2
			SH2R JK 2
			SH2R L 0 A_PlaySoundEx("weapons/sshotl", "auto", 0)
			SH2R LMNO 2
			SH2R PQRSTUVWX 2
			SH2R Y 0 A_PlaySoundEx("weapons/sshotc", "auto", 0)
			SH2R YZ 2
			SH22 ABCD 1
			SHT2 A 0 A_ReFire
			goto Ready
		Flash:
			SH2F A 2 bright A_Light1
			SH2F B 1 bright A_Light2
			SH2F B 0 bright A_Light0
			stop
	}
}
