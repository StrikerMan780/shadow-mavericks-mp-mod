// Ember code loosely based on Tormentor667, SolarSnowfall, and Ghastly_Dragon's Ember SFX
Actor RedEmber
{
 +Missile
 +DontSplash
 +NoBlockmap
 +NoGravity
 +ForceXYBillboard
 +ClientSideOnly
 +AllowClientSpawn
 -Solid
 Radius 3
 Height 3
 Speed 0
 RenderStyle Add
 Alpha 0.01
 Scale 0.03
 States
 {
  Spawn:
   TNT1 A 0
   TNT1 A 0 A_CheckSight("Death")
   REMB AAAA 1 Bright A_FadeIn(0.25)
   goto Move
  Move:
   REMB AAA 8 Bright A_Recoil(random(-1,1))
   goto Fade
  Fade:
   REMB A 1 Bright A_FadeOut(random(0.05, 0.1))
   loop
  Death:
   TNT1 A 1
   stop
 }
}
