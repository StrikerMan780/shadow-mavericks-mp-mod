actor BlankWeapon : Weapon
{
	Weapon.AmmoUse 0
	States
	{
		Deselect:
			TNT1 A 1 A_Lower
			loop
		Select:
			TNT1 A 1 A_Raise
			loop
		Ready:
			TNT1 A 1 A_WeaponReady(WRF_NOFIRE | WRF_DISABLESWITCH)
			loop
		Fire:
			TNT1 A 0
			goto Ready
	}
}
