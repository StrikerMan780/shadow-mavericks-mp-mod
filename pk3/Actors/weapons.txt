actor ClassWeaponGiver : CustomInventory
{
	Radius 20
	Height 16
	Speed 0
	Inventory.PickupMessage ""
	+INVENTORY.AUTOACTIVATE
	+INVENTORY.QUIET
	Args 300
	var int user_scale;
	states
	{
		Spawn:
			TNT1 A 0
			TNT1 A 0 A_CheckFlag("DROPPED", "Dropped")
			TNT1 A 0 A_SetUserVar("user_scale", 100)
			TNT1 A 0 A_JumpIf(1==1, "Item")
			wait
		Stay:
			"####" "#" 0 A_RailWait
			fail
		Remove:
			"####" "#" 0 A_RailWait
			stop
			
		CheckStatus:
			"####" "#" 0 A_CheckFlag("DROPPED", "FadeTimer")
			goto StateJump
		FadeTimer:
			"####" "#" 0 A_CountdownArg(0, "FadeAway")
			goto StateJump
		StateJump:
			"####" "#" 35 ACS_NamedExecuteAlways("PICKUPSTATE", 0)
			loop
			
		FadeAway:
			"####" "#" 1 A_FadeOut(0.005)
			loop
	}
}

actor BlankWeapon : Weapon
{
	Weapon.AmmoUse 0
	States
	{
		Deselect:
			TNT1 A 1 A_Lower
			loop
		Select:
			TNT1 A 1 A_Raise
			loop
		Ready:
			TNT1 A 1 A_WeaponReady(WRF_NOFIRE)
			loop
		Fire:
			TNT1 A 0
			goto Ready
	}
}
